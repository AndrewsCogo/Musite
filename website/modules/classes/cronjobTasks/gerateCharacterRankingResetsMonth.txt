global $TABLES_CONFIGS;
$this->queryStatement("TRUNCATE TABLE dbo.webRankingCharactersResetsMonth");
$findCharactersQ = $this->queryStatement("Select TOP 200 AccountID,Name,Class,cLevel,".COLUMN_PK_RANKING." as PkCount,CtlCode,".COLUMN_RESETS." as resets, ".COLUMN_MASTER_RESETS." as mresets, ".COLUMN_RESETS_WEEK." as resetsWeek,".COLUMN_RESETS_MONTH." as resetsMonth from ".DATABASE_CHARACTERS.".dbo.Character where (ctlcode < 8 or ctlcode is null) order by resetsMonth desc, resets desc");
while($findCharacters = mssql_fetch_object($findCharactersQ))
{
    $this->queryStatement("INSERT INTO dbo.webRankingCharactersResetsMonth (account,character,clevel,pkcount,resets,mresets,resetsweek,resetsmonth,class) VALUES ('{$findCharacters->AccountID}', '{$findCharacters->Name}', {$findCharacters->cLevel}, {$findCharacters->PkCount}, {$findCharacters->resets}, {$findCharacters->mresets}, {$findCharacters->resetsWeek}, {$findCharacters->resetsMonth}, {$findCharacters->Class})");
}
$this->addSymbol(42, $this->colors['black'], $this->colors['lightgreen']);  
$this->addStringImg(50, "Execução bem sucedida.", $this->colors['black']);
